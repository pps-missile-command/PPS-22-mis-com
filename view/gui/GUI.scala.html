31   /**
32    * Method that takes the grame obserable and map all the Point2D elements emitted due to the
33    * mouse click event into a [[Event.LaunchMissileTo]] events.
34    */
35   private val gameEvent: Observable[Event] =
36     frame
37       .getContentPane
38       .mouseObservable()
39       .map(p =>
40         Event
41           .LaunchMissileTo(

44               (p.y * World.height.toDouble) / ViewConstants.GUI_height.toDouble

46           )
