35     /***
36      * Calculate the new position of the vehicle
37      * @param vehicle Vehicle that have to be moved
38      * @return New position after moved time
39      */
40     def moveVehicle(vehicle: Plane): Point2D =
41         val distanceToMove = vehicle.velocity * deltaTime
42         val distanceToFinalPosition = vehicle.position <-> vehicle.destination
43         if distanceToMove >= distanceToFinalPosition then
44             vehicle.destination
45         else
46             vehicle.position --> (vehicle.direction * distanceToMove * (-1))
