39      * Calculate the new position of the vehicle
40      * @param vehicle Vehicle that have to be moved
41      * @return New position after moved time
42      */
43     def moveVehicle(vehicle: Plane): Point2D =
44         val distanceToMove = vehicle.velocity * deltaTime
45         val distanceToFinalPosition = vehicle.position <-> vehicle.destination
46         if distanceToMove >= distanceToFinalPosition then
47             vehicle.destination
48         else
49             vehicle.position --> (vehicle.direction * distanceToMove * (-1))
