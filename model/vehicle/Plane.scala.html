99      * @param vehicle Vehicle that have to be moved
100      * @return New position after moved time

102     def moveVehicle(vehicle: Plane): Point2D =
103         val distanceToMove = vehicle.velocity * deltaTime
104         val distanceToFinalPosition = vehicle.position <-> vehicle.destination
105         if distanceToMove >= distanceToFinalPosition then
106             vehicle.destination
107         else
108             vehicle.position --> (vehicle.direction * distanceToMove * (-1))
