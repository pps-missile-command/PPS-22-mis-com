11 /**
12  * This trait models a variation of the normal [[Missile]], that is the ZigZagMissile: the difference is
13  * that in this case the path is not linear from the start point towards the destination point, but a sequence
14  * subdestinations to reach, and when one is reached change the direction accordingly
15  */
16 trait ZigZagMissile(step: Int = 5, positions: LazyList[Point2D], to: Point2D):
17     missile: MissileImpl =>
18   /**
19    * This method check if the position is equals to the sub destination: the sub destination
20    * change every time the ZigZagMissile reach it, taking the next destination available into the list
21    * @return a boolean value true or false to models the equivalency
22    */
23     def subDestinationReached: Boolean = position ~= destination

25   /**
26    * This method overrides the same method defined in [[Missile]]: in this case we check if the current position is
27    * equal to the last position of the path, that is the final destination
28    * @return a boolean true or false
29    */
30     override def destinationReached: Boolean = position ~= to

32   /**
33    * Copy method that creates a new ZigZagMissile with the new properties specified

37    * @return the new ZigZagMissile
38    */
39     override def newMissile(life: LifePoint = missile.lifePoint, pos: Point2D = missile.position, _dt: DeltaTime = missile.dt): MissileImpl & ZigZagMissile & Scorable =
40       subDestinationReached match
