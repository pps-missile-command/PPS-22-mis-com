12 /**
13  * This trait models a variation of the normal [[Missile]], that is the ZigZagMissile: the difference is
14  * that in this case the path is not linear from the start point towards the destination point, but a sequence
15  * subdestinations to reach, and when one is reached change the direction accordingly
16  */
17 trait ZigZagMissile(step: Int = 5, positions: LazyList[Point2D], to: Point2D):
18     missile: MissileImpl =>
19   /**
20    * This method check if the position is equals to the sub destination: the sub destination
21    * change every time the ZigZagMissile reach it, taking the next destination available into the list
22    * @return a boolean value true or false to models the equivalency
23    */
24     def subDestinationReached: Boolean = position ~= destination
25 
26   /**
27    * This method overrides the same method defined in [[Missile]]: in this case we check if the current position is
28    * equal to the last position of the path, that is the final destination
29    * @return a boolean true or false
30    */
31     override def destinationReached: Boolean = position ~= to
32 
33   /**
34    * Copy method that creates a new ZigZagMissile with the new properties specified
35    * @param life The new life of the missile
36    * @param pos The updated position
37    * @param _dt The updated virtual delta time
38    * @return the new ZigZagMissile
39    */
40     override def newMissile(life: LifePoint = missile.lifePoint, pos: Point2D = missile.position, _dt: DeltaTime = missile.dt): MissileImpl & ZigZagMissile & Scorable =
41       subDestinationReached match
42         case true => ZigZagMissile.apply(missile, pos, step, positions.pop()._2, _dt, to)
43         case false => ZigZagMissile.apply(missile.copy(lifePoint = life), pos, step, positions, _dt, to)
