10 /**
11  * This trait models a variation of the normal [[Missile]], that is the ZigZagMissile: the difference is
12  * that in this case the path is not linear from the start point towards the destination point, but a sequence
13  * subdestinations to reach, and when one is reached change the direction accordingly
14  */
15 trait ZigZagMissile(step: Int = 5, positions: LazyList[Point2D], to: Point2D):
16     missile: MissileImpl =>
17   /**
18    * This method check if the position is equals to the sub destination: the sub destination
19    * change every time the ZigZagMissile reach it, taking the next destination available into the list
20    * @return a boolean value true or false to models the equivalency
21    */
22     def subDestinationReached: Boolean = position ~= destination
23 
24   /**
25    * This method overrides the same method defined in [[Missile]]: in this case we check if the current position is
26    * equal to the last position of the path, that is the final destination
27    * @return a boolean true or false
28    */
29     override def destinationReached: Boolean = position ~= to
30 
31   /**
32    * Copy method that creates a new ZigZagMissile with the new properties specified
33    * @param life The new life of the missile
34    * @param pos The updated position
35    * @param _dt The updated virtual delta time
36    * @return the new ZigZagMissile
37    */
38     override def newMissile(life: LifePoint = missile.lifePoint, pos: Point2D = missile.position, _dt: DeltaTime = missile.dt): MissileImpl & ZigZagMissile & Scorable =
39       subDestinationReached match
40         case true => ZigZagMissile.apply(missile, pos, step, positions.pop()._2, _dt, to)
41         case false => ZigZagMissile.apply(missile, pos, step, positions, _dt, to)
