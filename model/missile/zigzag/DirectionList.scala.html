37   def randomList(from: Point2D, to: Point2D, step: Int)(using Random): LazyList[Point2D] =
38     val d = (to <-> from) / step

46   /**
47    * This factory generates a LazyList with directions alternated (Right, Left, Right...), creating a zigzag path
48    * between two points
49    * @param from The starting point to create the path
50    * @param to The destination point
51    * @param step The number of direction changes to perform
52    * @param magnitude The magnitude of the rotated vectors used to traslate (on the left or right) the subpoints of the path
53    * @param maxWidth The max width whithin generate the path
54    * @param Random The Random object used to
55    * @return the new LazyList with the zigzag path
56    */
57   def apply(from: Point2D, to: Point2D, step: Int, magnitude: Int = defaultMagnitude, maxWidth: Double): LazyList[Point2D] =
58     val d = (to <-> from) / step
59     val v = ((to <--> from).normalize) * d
60     val stream = LazyList.iterate(0)(_ + 1)
61     val pointsStream = LazyList.iterate(from)( _ --> (-v) )
62     for
63       index <- stream.take(step)
64       if(index != 0)
65       point = pointsStream.take(index + 1).toList.last
66       direction = stream.take(index + 1).map {
67         i => (i % 2) match
68           case v if v == 0 => Right
69           case _ => Left
70       }.toList.last
71     yield (point --> (-v -|- (direction, magnitude))).filterMap(maxWidth)
