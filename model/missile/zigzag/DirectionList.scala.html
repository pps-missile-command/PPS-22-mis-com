26   /**
27    * This method creates a list with random zigzag behavior, this means that the directions are not alternated but
28    * generated randomically
29    * @param from The starting point to create the path
30    * @param to The destination point
31    * @param step The number of direction changes to perform
32    * @param Random The Random context element used to generate randomically the nth direction

34    */
35   def randomList(from: Point2D, to: Point2D, step: Int)(using Random): LazyList[Point2D] =
36     val d = (to <-> from) / step
37     val v = ((to <--> from).normalize) * d
38     DirectionList(from) {
39       i => i --> (-v)
40     } (_ != from) {
41       i => i --> (-v -|- Rand(Random))
42     }

44   /**
45    * This factory generates a LazyList with directions alternated (Right, Left, Right...), creating a zigzag path
46    * between two points
47    * @param from The starting point to create the path
48    * @param to The destination point
49    * @param step The number of direction changes to perform
50    * @param magnitude The magnitude of the rotated vectors used to traslate (on the left or right) the subpoints of the path
51    * @param maxWidth The max width whithin generate the path
52    * @param Random The Random object used to
53    * @return the new LazyList with the zigzag path
54    */
55   def apply(from: Point2D, to: Point2D, step: Int, magnitude: Int = defaultMagnitude, maxWidth: Double): LazyList[Point2D] =
56     val d = (to <-> from) / step
57     val v = ((to <--> from).normalize) * d
58     val stream = LazyList.iterate(0)(_ + 1)
59     val pointsStream = LazyList.iterate(from)( _ --> (-v) )
60     for
61       index <- stream.take(step)
62       if(index != 0)
63       point = pointsStream.take(index + 1).toList.last
64       direction = stream.take(index + 1).map {
65         i => (i % 2) match
66           case v if v == 0 => Right
67           case _ => Left
68       }.toList.last
69     yield (point --> (-v -|- (direction, magnitude))).filterMap(maxWidth)
