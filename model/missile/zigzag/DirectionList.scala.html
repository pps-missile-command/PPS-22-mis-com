27   def apply(from: Point2D, to: Point2D, step: Int)(using Random): LazyList[Point2D] =
28     val d = (to <-> from) / step
29     val v = ((to <--> from).normalize) * d
30     val stream = LazyList.iterate(0)(_ + 1)
31     val pointsStream = LazyList.iterate(from)( _ --> (-v) )
32     for
33       index <- stream.take(step)
34       if(index != 0)
35       point = pointsStream.take(index + 1).toList.last
36       direction = stream.take(index + 1).map {
37         i => (i % 2) match
38           case v if v == 0 => Right
39           case _ => Left
40       }.toList.last
41     yield point --> (-v -|- direction)
