95   /**
96    * If the damage to inflict is greater or equals than the current life points it returns
97    * a dead missile, otherwise it returns a new missile with its life points decremented by the damage inflicted
98    * @param damageToInflict
99    *  @return the object with the new health.

101   override def takeDamage(damageToInflict: LifePoint): Missile = damageToInflict match
102     case n if (currentLife - n) <= 0 => newMissile(life = lifePointDeath)
103     case n if (n > 0) => newMissile(life = currentLife - n)
104     case _ => this
