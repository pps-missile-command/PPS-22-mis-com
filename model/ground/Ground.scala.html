60     /***
61      * Method used for shooting the nearest turret to a determinate point.
62      * @param endPoint Point where the missile should explode
63      * @return Tuple containing the new updated ground and the missile shooted. Missile is null if all turrets were reloading
64      */
65     def shootMissile(endPoint: Point2D): Tuple2[Ground, Option[Missile]] =
66         def calculatePositions: List[Tuple2[Double, MissileBattery]] =
67             for battery <- missileBatteryAlive if battery.isReadyForShoot
68                 yield (battery.bottomLeft_Position <-> endPoint, battery)
69 
70         val batteriesReadyForShoot = calculatePositions
71         if(batteriesReadyForShoot.length > 0) then //only if there at least 1 turret ready for shoot, it will procede with shooting
72             val missilesBatteryInformations = batteriesReadyForShoot.minBy(_._1)._2.shootRocket(endPoint).get
73             val newTurrets = turrets.map( t => if (t == batteriesReadyForShoot.minBy(_._1)._2) missilesBatteryInformations._1 else t)
74             (Ground(cities, newTurrets) , Some(missilesBatteryInformations._2))
75         else
76             (Ground(cities, turrets), None)

84     def dealDamage(structure: Damageable, damageDealed: LifePoint, ground: Ground = this): Ground = structure match
85         case c: City => val newCities = ground.cities.map(o => if (o == c) o.takeDamage(damageDealed) else o)

88             Ground(ground.cities, newTurrets)
