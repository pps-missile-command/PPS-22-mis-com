62     /***

65      * @return Tuple containing the new updated ground and the missile shooted. Missile is null if all turrets were reloading

67     def shootMissile(endPoint: Point2D): Tuple2[Ground, Missile] =
68         def calculatePositions: List[Tuple2[Double, MissileBattery]] =
69             for battery <- missileBatteryAlive if battery.isReadyForShoot
70                 yield (battery.bottomLeft_Position <-> endPoint, battery)
71 
72         val batteriesReadyForShoot = calculatePositions
73         if(batteriesReadyForShoot.length > 0) then //only if there at least 1 turret ready for shoot, it will procede with shooting
74             val missilesBatteryInformations = batteriesReadyForShoot.minBy(_._1)._2.shootRocket(endPoint).get
75             val newTurrets = turrets.map( t => if (t == batteriesReadyForShoot.minBy(_._1)._2) missilesBatteryInformations._1 else t)
76             (Ground(cities, newTurrets) , missilesBatteryInformations._2)
77         else
78             (Ground(cities, turrets), null)
