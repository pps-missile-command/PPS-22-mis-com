58     /***
59      * Method used for shooting the nearest turret to a determinate point.
60      * @param endPoint Point where the missile should explode
61      * @return Tuple containing the new updated ground and the missile shooted. Missile is null if all turrets were reloading
62      */
63     def shootMissile(endPoint: Point2D): Tuple2[Ground, Missile] =
64         def calculatePositions: List[Tuple2[Double, MissileBattery]] =
65             for battery <- missileBatteryAlive if battery.isReadyForShoot
66                 yield (battery.position <-> endPoint, battery)
67 
68         val batteriesReadyForShoot = calculatePositions
69         if(batteriesReadyForShoot.length > 0) then //only if there at least 1 turret ready for shoot, it will procede with shooting
70             val missilesBatteryInformations = batteriesReadyForShoot.minBy(_._1)._2.shootRocket(endPoint).get
71             val newTurrets = turrets.map( t => if (t == batteriesReadyForShoot.minBy(_._1)._2) missilesBatteryInformations._1 else t)
72             (Ground(cities, newTurrets) , missilesBatteryInformations._2)
73         else
74             (Ground(cities, turrets), null)

82     def dealDamage(structure: Damageable, damageDealed: LifePoint, ground: Ground = this): Ground = structure match
83         case c: City => val newCities = ground.cities.map(o => if (o == c) o.takeDamage(damageDealed) else o)
84             Ground(newCities, ground.turrets)
85         case m: MissileBattery => val newTurrets = ground.turrets.map(o => if (o == m) o.takeDamage(damageDealed) else o)
86             Ground(ground.cities, newTurrets)
