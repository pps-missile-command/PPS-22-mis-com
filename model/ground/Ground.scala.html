61     /***
62      * Method used for shooting the nearest turret to a determinate point.
63      * @param endPoint Point where the missile should explode
64      * @return Tuple containing the new updated ground and the missile shooted. Missile is null if all turrets were reloading
65      */
66     def shootMissile(endPoint: Point2D): Tuple2[Ground, Option[Missile]] =
67         def calculatePositions: List[Tuple2[Double, MissileBattery]] =
68             for battery <- missileBatteryAlive if battery.isReadyForShoot
69                 yield (battery.bottomLeft_Position <-> endPoint, battery)
70 
71         val batteriesReadyForShoot = calculatePositions
72         if(batteriesReadyForShoot.length > 0) then //only if there at least 1 turret ready for shoot, it will procede with shooting
73             val missilesBatteryInformations = batteriesReadyForShoot.minBy(_._1)._2.shootRocket(endPoint).get
74             val newTurrets = turrets.map( t => if (t == batteriesReadyForShoot.minBy(_._1)._2) missilesBatteryInformations._1 else t)
75             (Ground(cities, newTurrets) , Some(missilesBatteryInformations._2))
76         else
77             (Ground(cities, turrets), None)
