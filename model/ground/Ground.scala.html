59     /***
60      * Method used for shooting the nearest turret to a determinate point.
61      * @param endPoint Point where the missile should explode
62      * @return Tuple containing the new updated ground and the missile shooted. Missile is null if all turrets were reloading
63      */
64     def shootMissile(endPoint: Point2D): Tuple2[Ground, Missile] =
65         def calculatePositions: List[Tuple2[Double, MissileBattery]] =
66             for battery <- missileBatteryAlive if battery.isReadyForShoot
67                 yield (battery.bottomLeft_Position <-> endPoint, battery)
68 
69         val batteriesReadyForShoot = calculatePositions
70         if(batteriesReadyForShoot.length > 0) then //only if there at least 1 turret ready for shoot, it will procede with shooting
71             val missilesBatteryInformations = batteriesReadyForShoot.minBy(_._1)._2.shootRocket(endPoint).get
72             val newTurrets = turrets.map( t => if (t == batteriesReadyForShoot.minBy(_._1)._2) missilesBatteryInformations._1 else t)
73             (Ground(cities, newTurrets) , missilesBatteryInformations._2)
74         else
75             (Ground(cities, turrets), null)

83     def dealDamage(structure: Damageable, damageDealed: LifePoint, ground: Ground = this): Ground = structure match
84         case c: City => val newCities = ground.cities.map(o => if (o == c) o.takeDamage(damageDealed) else o)

87             Ground(ground.cities, newTurrets)
