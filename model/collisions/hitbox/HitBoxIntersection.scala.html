20   def apply(hitBoxes: HitBox*)(using Distance): HitBox =
21     val intersection = hitBoxes match
22       case Seq() => HitBoxEmpty
23       case Seq(hitBox) => hitBox
24       case _ => HitBoxIntersection(hitBoxes)

29       case _ if intersection.yMax.get < intersection.yMin.get => HitBoxEmpty
30       case _ if intersection.area.isEmpty => HitBoxEmpty
